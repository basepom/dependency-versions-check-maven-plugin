208     /**
209      * List of resolvers to apply specific strategies to dependencies.
210      *
211      * <pre>
212      * &lt;resolvers&gt;
213      *   &lt;resolver&gt;
214      *     &lt;strategy&gt;...&lt;strategy&gt;
215      *     &lt;includes&gt;
216      *       &lt;include&gt;...&lt;include&gt;
217      *     &lt;includes&gt;
218      *   &lt;resolver&gt;
219      * &lt;resolvers&gt;
220      * </pre>
221      *
222      * A resolver maps a specific strategy to a list of includes.
223      * The include syntax is <code>[group-id]:[artifact-id]</code> where each pattern segment

226      * The plugin includes some default strategies: <code>apr</code>, <code>default</code>, <code>single-digit</code> and <code>two-digits-backward-compatible</code>.

246     public void execute()
247             throws MojoExecutionException, MojoFailureException
248     {
249         try {
250             for (VersionCheckExcludes exclusion : exclusions) {
251                 checkState(exclusion.isValid(), "Invalid exclusion specification: '%s'", exclusion);
252             }
253 
254             checkState(!Strings.nullToEmpty(scope).trim().isEmpty() && VALID_SCOPES.contains(scope), "Scope '%s' is invalid", scope);
255 
256             if (skip) {
257                 LOG.report(quiet, "Skipping plugin execution");
258                 return;
259             }
260 
261             if (!includePomProjects && "pom".equals(project.getPackaging())) {
262                 LOG.report(quiet, "Ignoring POM project");
263                 return;
264             }
265 
266             LOG.debug("Starting %s mojo run!", this.getClass().getSimpleName());
267 
268             this.strategyCache = new StrategyCache(strategyProvider, resolvers, defaultStrategy);
269 
270             final ScopeLimitingFilter scopeFilter = createScopeFilter();
271             final DependencyMap rootDependencyMap = new DependencyMapBuilder(this).mapProject(project, scopeFilter);
272 
273             try (DependencyTreeResolverDependencyTreeResolver">DependencyTreeResolver dependencyTreeResolver = new DependencyTreeResolver(this, rootDependencyMap)) {
274                 final ImmutableSetMultimap<QualifiedName, VersionResolutionCollection> resolutionMap = dependencyTreeResolver.computeResolutionMap(project, scopeFilter);
275                 doExecute(resolutionMap, rootDependencyMap);
276             }
277         }
278         catch (MojoExecutionException | MojoFailureException e) {
279             throw e;
280         }
281         catch (Exception e) {
282             throw new MojoExecutionException("While running mojo: ", e);
283         }
284         finally {
285             LOG.debug("Ended %s mojo run!", this.getClass().getSimpleName());
286         }
287     }
